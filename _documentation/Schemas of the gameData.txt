Schemas of the gameData

globalData = {
    mainGame: gameData
}

gameData = {
    players: {
        "id1": Player,
        "id2": Player
    },
    player_keys: [id1, id2],
    canDuelAction: Bool,
    isGameLaunched: Bool,
    currentRoundStartTime: Date
}

Player = {
    hp: Integer,
    currentCards: [Card, Card, Card],
    remainingCardsStack: [Card, Card, ...],
    action: 'SHIELD' | 'ATTACK'
    actionCardIndex: 0 | 1 | 2
}

Card = {
    value: Integer,
    element: 'ROCK' | 'PAPER' | 'SCISSOR'
}
